CREATE TABLE [Besitzer] (
  [Besitzer_ID] int,
  [Vorname] VARCHAR(50),
  [Nachname] VARCHAR(50),
  [Besitzbeginn] date,
  [Besitzende] date
);

CREATE TABLE [Fahrzeugtyp] (
  [Fahrzeugart] VARCHAR(50),
  [Model] VARCHAR(50),
  [Wert] FLOAT,
  [Farbe] VARCHAR(30),
  [Fahrzeugtyp_ID] int
);

CREATE TABLE [Eigentuemer] (
  [Eigentuemer_ID] int,
  [Vorname] VARCHAR(50),
  [Nachname] VARCHAR(50)
);

CREATE TABLE [Kennzeichen] (
  [Kennzeichen_ID] int,
  [Besitzer] int,
  [Fahrzeugtyp] int,
  [Kennzeichen] VARCHAR(50),
  [Land] VARCHAR(50),
  [Eigentuemer] int,
  CONSTRAINT [FK_Kennzeichen.Besitzer]
    FOREIGN KEY ([Besitzer])
      REFERENCES [Besitzer]([Besitzer_ID]),
  CONSTRAINT [FK_Kennzeichen.Fahrzeugtyp]
    FOREIGN KEY ([Fahrzeugtyp])
      REFERENCES [Fahrzeugtyp]([Fahrzeugtyp_ID]),
  CONSTRAINT [FK_Kennzeichen.Eigentuemer]
    FOREIGN KEY ([Eigentuemer])
      REFERENCES [Eigentuemer]([Eigentuemer_ID])
);

CREATE TABLE [Besitzer Eigentuemer Verwandschaft] (
  [Besitzer] int,
  [Eigentuemer] int,
  CONSTRAINT [FK_Besitzer Eigentuemer Verwandschaft.Eigentuemer]
    FOREIGN KEY ([Eigentuemer])
      REFERENCES [Eigentuemer]([Eigentuemer_ID]),
  CONSTRAINT [FK_Besitzer Eigentuemer Verwandschaft.Besitzer]
    FOREIGN KEY ([Besitzer])
      REFERENCES [Besitzer]([Besitzer_ID])
);
