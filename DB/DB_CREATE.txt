CREATE TABLE [KontoInhaber] (
  [KontoInhaber_ID] INTEGER NOT NULL,
  [Vorname] VARCHAR(50) NOT NULL,
  [Nachname] VARCHAR(50) NOT NULL,
  PRIMARY KEY (KontoInhaber_ID)
);

CREATE TABLE [Konto] (
  [Konto_ID] INTEGER NOT NULL,
  [Inhaber] INTEGER NOT NULL,
  [Kontotyp] VARCHAR(20),
  [Kontonummer] VARCHAR(50) NOT NULL,
  [Land] VARCHAR(3) NOT NULL,
  [IBAN] VARCHAR(34) NOT NULL,
  [Betrag] DECIMAL NOT NULL,
  CONSTRAINT [FK_Konto.Inhaber]
    FOREIGN KEY ([Inhaber])
      REFERENCES [KontoInhaber]([KontoInhaber_ID]),
  PRIMARY KEY (Konto_ID)
);

CREATE TABLE [Projekt] (
  [ProjektID] INTEGER NOT NULL,
  [Name] VARCHAR(50) NOT NULL,
  [Eingeplantes_Budget] DECIMAL NOT NULL,
  [Eingetroffenes_Budget] DECIMAL NOT NULL,
  [KontoID] INTEGER NOT NULL,
  CONSTRAINT [FK_Konto.Konto_ID]
    FOREIGN KEY ([KontoID])
      REFERENCES [Konto]([Konto_ID]),
  PRIMARY KEY (ProjektID)
);
